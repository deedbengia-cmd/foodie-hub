function addFastFoodToCart(name, qtyId){ let quantity = parseInt(document.getElementById(qtyId).value); let priceList = { "Chow Noodles":100, "Momos":100, "Thukpa":100, "Fried Rice":100 }; let price = priceList[name]; let itemIndex = cart.findIndex(item => item.name === name && item.price === price); if(itemIndex > -1){ cart[itemIndex].quantity += quantity; } else { cart.push({name:name, price:price, quantity:quantity}); } localStorage.setItem("cart", JSON.stringify(cart)); alert(name + " added to cart!"); }
